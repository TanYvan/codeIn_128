12:   def get_tax(a)
13:     #参数a为应发金额
14: #    税款计算
15: #    酬金小于800不扣税；
16: #    酬金大于或等于800和小于等于4000的计算式：(酬金-800)*0.2
17: #
18: #    酬金大于4000以上的，先计算 应纳税所得额，即：应纳税所得额=酬金×（1-20%）
19: #    然后按照所得额计算税款,方法如下：
20: #    所得额在20000元以下：应纳税额=应纳所得税额×20%；
21: #    所得额在20000～50000元的：应纳税额=应纳税所得额×30%-2000；
22: #    所得额在50000元以上的：应纳税额=应纳税所得额×40%-7000。
23:     r=0
24:     if a<800
25:       r=0
26:     elsif a>=800 and a<=4000
27:       r=(a-800) * 0.2
28:     else
29:       c=a * 0.8
30:       if c<20000
31:         r=c * 0.2
32:       elsif c>20000 and c<50000
33:         r=c * 0.3 - 2000
34:       else
35:         r= c * 0.4 - 7000
36:       end
37:     end
38:     return r
39:   end
